> Implement an algorithm to determine if a string has all unique characters. 

* Note: Extra condition is needed if input string contains spaces. e.g. 'abc def ghi'.

```python
'''
Extra data structure dict. aka. hashtable
'''

def IsUnique(str):
    unique = {}
    for char in str:
        if char in unique:
            print("{} is NOT unique".format(str))
            return
        else:
            unique[char] = True
    print("{} is unique".format(str))
    
if __name__ == '__main__':
    IsUnique("dallas")
    IsUnique("hamilton")
    IsUnique("stamford")
    IsUnique("baltimore")
    
# dallas is NOT unique
# hamilton is unique
# stamford is unique
# baltimore is unique

```

#### What if you cannot use additional data structures?

```python
'''
Reasoning:
if two numbers are same, the bitwise & for the same numbers should be always zero.
'''

def IsUniqueBitOps(str):
    checker = 0
    for char in str:
        val = ord(char)
        lshift = 1 << val
        if checker & lshift > 0:
            print("{} is NOT unique".format(str))
            return
        checker |= lshift
    print("{} is unique".format(str))
    
if __name__ == '__main__':
    IsUniqueBitOps("allen")
    IsUniqueBitOps("richmond")
    IsUniqueBitOps("beaumont")
    IsUniqueBitOps("chandler")

# allen is NOT unique
# richmond is unique
# beaumont is unique
# chandler is unique

```
