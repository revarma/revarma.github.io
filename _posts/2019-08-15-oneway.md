> There are three types of edits that can be performed on strings: insert a character, remove a character, or replace a character. Given two strings, write a function to check if they are one edit (or zero edits) away.

> EXAMPLE: 
* ` pale, ple -> true `
* ` pales, pale -> true `
* ` pale, bale -> true `
* ` pale, bake -> false `


```python
def OneAway(str1, str2):
    if len(str1) - len(str2) > 1 or len(str1) - len(str2) < -1:
        print('{}, {} - {}'.format(str1, str2, False))
        return

    str1_dict = {}
    for char in str1:
        if char in str1_dict:
            str1_dict[char] += 1
        else:
            str1_dict[char] = 1

    str2_dict = {}
    for char in str2:
        if char in str2_dict:
            str2_dict[char] += 1
        else:
            str2_dict[char] = 1

    # Assuming str1 length is always bigger or equal to str2 length
    # if add another condition to compare lengths.
    edits_reqiuired = 0
    for k, v in str1_dict.items():
        if k not in str2_dict:
            edits_reqiuired += 1
            if edits_reqiuired > 1:
                print('{}, {} - {}'.format(str1, str2, False))
                return
        else:
            if v > str2_dict[k]:
                edits_reqiuired += 1
                if edits_reqiuired > 1:
                    print('{}, {} - {}'.format(str1, str2, False))
                    return
    print('{}, {} - {}'.format(str1, str2, True))


if __name__ == '__main__':
    OneAway('pale', 'ple')
    OneAway('pales', 'pale')
    OneAway('pale', 'bale')
    OneAway('pale', 'bake')
    
    
# pale, ple - True
# pales, pale - True
# pale, bale - True
# pale, bake - False
```
